50  final Map<String, String> map = new HashMap<>();
51  try (BufferedReader reader = new BufferedReader(new InputStreamReader(in))) {
52  String line, key, value;
53  int eqPos;
54  while ((line = reader.readLine()) != null) {
55  line = line.trim();
56  if (line.startsWith("#")) continue;

96  public static void replaceShellExports (File f, Map<String, String> exports) throws IOException {
97 
98  // validate -- no quote chars allowed for security reasons
99  for (String key : exports.keySet()) {
100  if (key.contains("\"") || key.contains("\'")) throw new IllegalArgumentException("replaceShellExports: name cannot contain a quote character: "+key);
101  String value = exports.get(key);
102  if (value.contains("\"") || value.contains("\'")) throw new IllegalArgumentException("replaceShellExports: value for "+key+" cannot contain a quote character: "+value);
103  }
104 
105  // read entire file as a string
106  final String contents = FileUtil.toString(f);
107 
108  // walk file line by line and look for replacements to make, overwrite file.
109  final Set<String> replaced = new HashSet<>(exports.size());
110  try (Writer w = new FileWriter(f)) {
111  for (String line : contents.split("\n")) {
112  line = line.trim();
113  boolean found = false;
114  for (String key : exports.keySet()) {
115  if (!line.startsWith("#") && line.matches("^\\s*export\\s+" + key + "\\s*=.*")) {
116  w.write("export " + key + "=\"" + exports.get(key) + "\"");
117  replaced.add(key);
118  found = true;
119  break;
120  }
121  }
122  if (!found) w.write(line);
123  w.write("\n");
124  }
125 
126  for (String key : exports.keySet()) {
127  if (!replaced.contains(key)) {
128  w.write("export "+key+"=\""+exports.get(key)+"\"\n");
129  }
130  }
131  }
132  }

152  public static CommandResult exec (Command command) throws IOException {
153 
154  final DefaultExecutor executor = new DefaultExecutor();
155 
156  final ByteArrayOutputStream outBuffer = new ByteArrayOutputStream();
157  OutputStream out = command.hasOut() ? new TeeOutputStream(outBuffer, command.getOut()) : outBuffer;
158  if (command.isCopyToStandard()) out = new TeeOutputStream(out, System.out);
159 
160  final ByteArrayOutputStream errBuffer = new ByteArrayOutputStream();
161  OutputStream err = command.hasErr() ? new TeeOutputStream(errBuffer, command.getErr()) : errBuffer;
162  if (command.isCopyToStandard()) err = new TeeOutputStream(err, System.err);
163 
164  final ExecuteStreamHandler handler = new PumpStreamHandler(out, err, command.getInputStream());
165  executor.setStreamHandler(handler);
166 
167  if (command.hasDir()) executor.setWorkingDirectory(command.getDir());
168  executor.setExitValues(command.getExitValues());
169 
170  try {
171  final int exitValue = executor.execute(command.getCommandLine(), command.getEnv());
172  return new CommandResult(exitValue, outBuffer, errBuffer);
173 
174  } catch (Exception e) {
175  final String stdout = outBuffer.toString().trim();
176  final String stderr = errBuffer.toString().trim();
177  log.error("exec("+command.getCommandLine()+"): " + e
178  + (stdout.length() > 0 ? "\nstdout="+ellipsis(stdout, 1000) : "")
179  + (stderr.length() > 0 ? "\nstderr="+ellipsis(stderr, 1000) : ""));
180  return new CommandResult(e, outBuffer, errBuffer);
181  }
182  }
