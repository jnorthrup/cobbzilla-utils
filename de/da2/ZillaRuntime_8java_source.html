104  public static boolean empty(Object o) {

106  if (o instanceof String) return o.toString().length() == 0;
107  if (o instanceof Collection) return ((Collection)o).isEmpty();
108  if (o instanceof Map) return ((Map)o).isEmpty();
109  if (o instanceof JsonNode) {

111  if (o instanceof ArrayNode) return ((ArrayNode) o).size() == 0;
112  final String json = ((JsonNode) o).textValue();
113  return json == null || json.length() == 0;

115  if (o instanceof Iterable) return !((Iterable)o).iterator().hasNext();
116  if (o instanceof File) {

118  return !f.exists() || f.length() == 0 || (f.isDirectory() && list(f).length == 0);
