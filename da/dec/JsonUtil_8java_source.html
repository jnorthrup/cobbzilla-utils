276  public static List<JsonNode> findNodePath(JsonNode node, String path) throws IOException {
277 
278  final List<JsonNode> nodePath = new ArrayList<>();
279  nodePath.add(node);
280  if (empty(path)) return nodePath;
281  final List<String> pathParts = tokenize(path);
282 
283  for (String pathPart : pathParts) {
284  int index = -1;
285  int bracketPos = pathPart.indexOf("[");
286  int bracketClosePos = pathPart.indexOf("]");
287  boolean isEmptyBrackets = false;
288  if (bracketPos != -1 && bracketClosePos != -1 && bracketClosePos > bracketPos) {
289  if (bracketClosePos == bracketPos+1) {
290  // ends with [], they mean to append
291  isEmptyBrackets = true;
292  } else {
293  index = Integer.parseInt(pathPart.substring(bracketPos + 1, bracketClosePos));
294  }
295  pathPart = pathPart.substring(0, bracketPos);
296  }
297  if (!empty(pathPart)) {
298  node = node.get(pathPart);
299  if (node == null) {
300  nodePath.add(MISSING);
301  return nodePath;
302  }
303  nodePath.add(node);
304 
305  } else if (nodePath.size() > 1) {
306  return die("findNodePath: invalid path: "+path);
307  }
308  if (index != -1) {
309  node = node.get(index);
310  nodePath.add(node);
311 
312  } else if (isEmptyBrackets) {
313  nodePath.add(MISSING);
314  return nodePath;
315  }
316  }
317  return nodePath;

320  public static List<String> tokenize(String path) {
321  final List<String> pathParts = new ArrayList<>();
322  final StringTokenizer st = new StringTokenizer(path, ".'", true);
323  boolean collectingQuotedToken = false;
324  StringBuffer pathToken = new StringBuffer();
325  while (st.hasMoreTokens()) {
326  final String token = st.nextToken();
327  if (token.equals("'")) {
328  collectingQuotedToken = !collectingQuotedToken;
329 
330  } else if (collectingQuotedToken) {
331  pathToken.append(token);
332 
333  } else if (token.equals(".") && pathToken.length() > 0) {
334  pathParts.add(pathToken.toString());
335  pathToken = new StringBuffer();
336 
337  } else {
338  pathToken.append(token);
339  }
340  }
341  if (collectingQuotedToken) throw new IllegalArgumentException("Unterminated single quote in: "+path);
342  if (pathToken.length() > 0) pathParts.add(pathToken.toString());
343  return pathParts;

354  public static ObjectNode replaceNode(ObjectNode document, String path, String replacement) throws Exception {
355 
356  final String simplePath = path.contains(".") ? path.substring(path.lastIndexOf(".")+1) : path;
357  Integer index = null;
358  if (simplePath.contains("[")) {
359  index = Integer.parseInt(simplePath.substring(simplePath.indexOf("[")+1, simplePath.indexOf("]")));
360  }

363  throw new IllegalArgumentException("path not found: "+path);

367  if (index != null) {
