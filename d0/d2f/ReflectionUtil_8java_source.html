443  final boolean isMap = dest instanceof Map;
444  try {
445  if (src instanceof Map) copyFromMap(dest, (Map<String, Object>) src, exclude);
446 
447  checkGetter:
448  for (Method getter : src.getClass().getMethods()) {
449  // only look for getters on the source object (methods with no arguments that have a return value)
450  final Class<?>[] types = getter.getParameterTypes();
451  if (types.length != 0) continue;
452  if (getter.getReturnType().equals(Void.class)) continue;;
453 
454  // and it must be named appropriately
455  final String fieldName = fieldName(getter.getName());
456  if (fieldName == null || ArrayUtils.contains(exclude, fieldName)) continue;
457 
458  // if specific fields were given, it must be one of those
459  if (fields != null && !ArrayUtils.contains(fields, fieldName)) continue;
460 
461  // getter must not be marked @Transient
462  if (isIgnored(src, fieldName, getter)) continue;

464  // what would the setter be called?

466  if (setterName == null) continue;

473  } catch (Exception e) {
474  log.debug("copy: setter not found: " + setterName);
475  continue;
476  }
477  }
478 
479  // do not copy null fields (should this be configurable?)
480  final Object srcValue = getter.invoke(src);
481  if (srcValue == null) continue;
482 
483  // does the dest have a getter? if so grab the current value
484  Object destValue = null;
485  try {

488  } else {

503  }

895  public static <T> Class<T> getTypeParameter(Class<?> klass, Class<? super T> bound) {
896  Type t = checkNotNull(klass);
897  while (t instanceof Class<?>) {
898  t = ((Class<?>) t).getGenericSuperclass();
899  }
900  /* This is not guaranteed to work for all cases with convoluted piping
901  * of type parameters: but it can at least resolve straight-forward
902  * extension with single type parameter (as per [Issue-89]).
903  * And when it fails to do that, will indicate with specific exception.
904  */
905  if (t instanceof ParameterizedType) {
906  // should typically have one of type parameters (first one) that matches:
907  for (Type param : ((ParameterizedType) t).getActualTypeArguments()) {
908  if (param instanceof Class<?>) {
909  final Class<T> cls = determineClass(bound, param);
910  if (cls != null) { return cls; }
911  }
912  else if (param instanceof TypeVariable) {
913  for (Type paramBound : ((TypeVariable<?>) param).getBounds()) {
914  if (paramBound instanceof Class<?>) {
915  final Class<T> cls = determineClass(bound, paramBound);
916  if (cls != null) { return cls; }
917  }
918  }
919  }
920  }
921  }
922  return die("Cannot figure out type parameterization for " + klass.getName());
